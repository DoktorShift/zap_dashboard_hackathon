The index.html loads the nostr-login script.
Auth.vue initializes nostr-login and checks for existing sessions.
When a user clicks "Connect with Nostr", Auth.vue dispatches an event to launch the nostr-login modal.
The nostr-login library handles the interaction with the user's Nostr extension or private key.
Upon successful authentication, nostr-login dispatches an nlAuth event.
Auth.vue (or nostrAuth.js if used as an intermediary) listens for this event, extracts the user's public key, and optionally fetches their full Nostr profile from relays.
The useNostrSupabaseLogin composable is then used to upsert this user data (public key and profile) into your Supabase users table. ( We only work with organized local storage!!! no supabase)
The auth Pinia store updates its state with the authenticated user's information and persists the public key in local storage.
The application then redirects the user to the main content.

We need to store full nostr meta profile data of this suer in local storage.

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '@/stores/auth'
import { useNostrSupabaseLogin } from '@/composables/useNostrSupabaseLogin'
import { getProfileFromPubkey, updateLastSeenWithProfile, syncUserProfile } from '@/utils/profileSync.ts'

const router = useRouter()
const authStore = useAuthStore()
const { loginOrSignup, isProcessing, error } = useNostrSupabaseLogin()

const loginStatus = ref('')
const wasLoggedInBefore = ref(false)
const showLoginButton = ref(true)

const handleLogin = async (pubkey, fetchProfile = true) => {
  try {
    console.log('🔐 Handling login for pubkey:', pubkey.slice(0, 8) + '...')
    loginStatus.value = 'Connecting to Nostr network...'
    
    let profile = {}
    if (fetchProfile) {
      loginStatus.value = 'Fetching your profile...'
      profile = await getProfileFromPubkey(pubkey) || {}
    }

    loginStatus.value = 'Syncing with database...'
    const user = await loginOrSignup(pubkey, profile)
    
    // Set user in auth store
    await authStore.setNostrUser(pubkey, profile)
    
    // Update last seen
    await updateLastSeenWithProfile(pubkey, profile)
    
    console.log('✅ Login successful, redirecting...')
    loginStatus.value = 'Success! Redirecting...'
    
    setTimeout(() => {
      router.push('/games')
    }, 1000)
    
  } catch (err) {
    console.error('❌ Login failed:', err)
    loginStatus.value = `Login failed: ${err.message}`
  }
}

const handleSignup = async (pubkey) => {
  try {
    console.log('📝 Handling signup for pubkey:', pubkey.slice(0, 8) + '...')
    await handleLogin(pubkey, true) // Signup is the same as login in Nostr
  } catch (err) {
    console.error('❌ Signup failed:', err)
    loginStatus.value = `Signup failed: ${err.message}`
  }
}

const initializeNostrLogin = () => {
  let scriptLoaded = false

  // Check if user was logged in before
  const checkExistingLogin = async () => {
    try {
      if (window.nostr) {
        const pk = await window.nostr.getPublicKey()
        if (pk) {
          wasLoggedInBefore.value = true
          showLoginButton.value = false
          loginStatus.value = 'Restoring session...'
          handleLogin(pk, false)
        }
      }
    } catch (error) {
      console.log('No existing login found')
      wasLoggedInBefore.value = false
      showLoginButton.value = true
    }
  }

  // Load nostr-login script only once
  if (!document.querySelector('script[src*="nostr-login"]') && !scriptLoaded) {
    const script = document.createElement('script')
    script.src = 'https://www.unpkg.com/nostr-login@latest/dist/unpkg.js'
    script.setAttribute('data-theme', 'ocean')
    script.setAttribute('data-perms', 'sign_event:1,sign_event:0')
    script.setAttribute('data-title', 'Nostr Gaming Hub')
    script.setAttribute('data-description', 'Connect your Nostr identity to access minigames')
    script.setAttribute('data-no-banner', 'false')
    
    script.onload = () => {
      scriptLoaded = true
      checkExistingLogin()
    }
    
    document.head.appendChild(script)
  } else {
    checkExistingLogin()
  }
}

// Auth event handler
const handleAuthEvent = async (e) => {
  console.log('Auth event:', e.detail)
  try {
    if (window.nostr) {
      const pk = await window.nostr.getPublicKey()
      if (e.detail.type === 'login') {
        handleLogin(pk)
      } else if (e.detail.type === 'signup') {
        handleSignup(pk)
      }
      // Note: Logout events are now handled centrally in App.vue
    }
  } catch (error) {
    console.error('Auth event error:', error)
    loginStatus.value = `Authentication error: ${error.message}`
  }
}

const manualLogin = () => {
  try {
    if (window.nostr_login && window.nostr_login.launch) {
      window.nostr_login.launch()
    } else {
      // Fallback: dispatch login event
      document.dispatchEvent(new Event('nlLaunch'))
    }
  } catch (error) {
    console.error('Manual login error:', error)
    loginStatus.value = 'Please install a Nostr extension like nos2x or Alby'
  }
}

onMounted(() => {
  // Add event listeners for nostr-login
  document.addEventListener('nlAuth', handleAuthEvent)
  document.addEventListener('nlLogin', handleAuthEvent)  
  document.addEventListener('nlSignup', handleAuthEvent)
  
  // Initialize nostr-login
  initializeNostrLogin()
})

onUnmounted(() => {
  // Cleanup event listeners
  document.removeEventListener('nlAuth', handleAuthEvent)
  document.removeEventListener('nlLogin', handleAuthEvent)
  document.removeEventListener('nlSignup', handleAuthEvent)
})
</script>

import { defineStore } from 'pinia'
import { ref, computed, readonly } from 'vue'
import { supabase } from '@/lib/supabase'
import { updateLastSeenWithProfile } from '@/utils/profileSync.ts'
import * as nip19 from 'nostr-tools/nip19'
import * as Sentry from '@sentry/vue'

export const useAuthStore = defineStore('auth', () => {
  const user = ref(null)
  const profile = ref(null)
  const loading = ref(true)
  const error = ref(null)
  const authLoading = ref(false)
  const pubkey = ref(null)

  const isAuthenticated = computed(() => !!user.value)
  const isNostrUser = computed(() => !!pubkey.value)

  const initialize = async () => {
    try {
      loading.value = true
      
      console.log('🚀 Initializing auth store...')
      
      // Check for stored pubkey
      const storedPubkey = localStorage.getItem('userPubkey')
      if (storedPubkey) {
        console.log('Found stored pubkey, loading user data...')
        await loadUserByPubkey(storedPubkey)
      }
      
    } catch (err) {
      console.error('❌ Auth initialization error:', err)
      error.value = err.message
      Sentry.captureException(err)
    } finally {
      loading.value = false
    }
  }

  const loadUserByPubkey = async (userPubkey) => {
    try {
      console.log('📖 Loading user by pubkey:', userPubkey.slice(0, 8) + '...')
      
      // Fetch user from the users table (which includes profile data)
      const { data: userData, error: userError } = await supabase
        .from('users')
        .select('*')
        .eq('pubkey', userPubkey)
        .maybeSingle()

      if (userError && userError.code !== 'PGRST116') {
        throw userError
      }

      if (userData) {
        pubkey.value = userPubkey
        user.value = userData
        profile.value = userData // User table contains profile data
        
        // Only update last_seen_at timestamp without overwriting profile data
        try {
          const { error: lastSeenError } = await supabase
            .from('users')
            .update({ last_seen_at: new Date().toISOString() })
            .eq('pubkey', userPubkey)

          if (lastSeenError) {
            console.warn('Failed to update last seen timestamp:', lastSeenError)
          } else {
            console.log('✅ Updated last seen timestamp')
          }
        } catch (lastSeenError) {
          console.warn('Failed to update last seen timestamp:', lastSeenError)
        }
        
        console.log('✅ User loaded successfully:', userData.name || 'Nostr User')
      } else {
        console.log('⚠️ No user found for pubkey')
        clearAuth()
      }
      
    } catch (err) {
      console.error('❌ Failed to load user:', err)
      clearAuth()
      throw err
    }
  }

  const setNostrUser = async (nostrPubkey, nostrProfile = {}) => {
    try {
      authLoading.value = true
      console.log('🔐 Setting Nostr user:', nostrPubkey.slice(0, 8) + '...')

      // Generate npub using nostr-tools
      let npub = null
      try {
        npub = nip19.npubEncode(nostrPubkey)
      } catch (e) {
        console.warn('Failed to generate npub:', e)
      }
      
      // Check if we have meaningful profile data to update
      const hasProfileData = nostrProfile && (
        nostrProfile.name || 
        nostrProfile.display_name || 
        nostrProfile.picture || 
        nostrProfile.about || 
        nostrProfile.banner || 
        nostrProfile.nip05 || 
        nostrProfile.lud16 || 
        nostrProfile.website
      )
      
      // Create/update user in database (let database auto-generate UUID for id)
      const userData = {
        pubkey: nostrPubkey,
        npub: npub,
        updated_at: new Date().toISOString(),
        last_seen_at: new Date().toISOString()
      }
      
      // Only add profile fields if we have meaningful data
      if (hasProfileData) {
        Object.assign(userData, {
          name: nostrProfile.name || nostrProfile.display_name || 'Nostr User',
          display_name: nostrProfile.display_name || nostrProfile.name || '',
          about: nostrProfile.about || '',
          picture: nostrProfile.picture || null,
          banner: nostrProfile.banner || null,
          nip05: nostrProfile.nip05 || null,
          lud16: nostrProfile.lud16 || null,
          website: nostrProfile.website || null
        })
      }

      const { data: upsertedUser, error: userError } = await supabase
        .from('users')
        .upsert(userData, {
          onConflict: 'pubkey',
          ignoreDuplicates: false
        })
        .select()
        .single()

      if (userError) {
        throw new Error(`Failed to create/update user: ${userError.message}`)
      }

      // Set auth state
      pubkey.value = nostrPubkey
      user.value = upsertedUser || userData
      profile.value = upsertedUser || userData
      
      // Store pubkey for persistence
      localStorage.setItem('userPubkey', nostrPubkey)
      
      console.log('✅ Nostr user set successfully')
      return upsertedUser || userData
      
    } catch (err) {
      console.error('❌ Failed to set Nostr user:', err)
      error.value = err.message
      Sentry.captureException(err)
      throw err
    } finally {
      authLoading.value = false
    }
  }

  const signOut = async () => {
    try {
      console.log('🚪 Signing out...')
      
      clearAuth()
      
      // Clear nostr-login session if available
      if (window.nostr_login && window.nostr_login.logout) {
        window.nostr_login.logout()
      }
      
      console.log('✅ Signed out successfully')
    } catch (err) {
      console.error('❌ Sign out error:', err)
      error.value = err.message
    }
  }

  const clearAuth = () => {
    user.value = null
    profile.value = null
    pubkey.value = null
    error.value = null
    
    // Clear localStorage
    localStorage.removeItem('userPubkey')
    
    console.log('🗑️ Auth state cleared')
  }

  const updateProfile = async (updates) => {
    try {
      if (!pubkey.value) {
        throw new Error('No authenticated user')
      }

      console.log('📝 Updating profile...')
      
      // If updating npub, ensure it's properly encoded
      if (updates.npub && !updates.npub.startsWith('npub1')) {
        try {
          updates.npub = nip19.npubEncode(pubkey.value)
        } catch (e) {
          console.warn('Failed to encode npub in update:', e)
          delete updates.npub // Remove invalid npub from updates
        }
      }
      
      const { data: updatedUser, error } = await supabase
        .from('users')
        .update({
          ...updates,
          updated_at: new Date().toISOString()
        })
        .eq('pubkey', pubkey.value)
        .select()
        .single()

      if (error) {
        throw error
      }

      // Update local state
      user.value = updatedUser
      profile.value = updatedUser
      
      console.log('✅ Profile updated successfully')
      return updatedUser
      
    } catch (err) {
      console.error('❌ Failed to update profile:', err)
      error.value = err.message
      Sentry.captureException(err)
      throw err
    }
  }

  return {
    // State
    user: readonly(user),
    profile: readonly(profile),
    loading: readonly(loading),
    error: readonly(error),
    authLoading: readonly(authLoading),
    pubkey: readonly(pubkey),
    
    // Computed
    isAuthenticated,
    isNostrUser,
    
    // Actions
    initialize,
    setNostrUser,
    signOut,
    clearAuth,
    updateProfile,
    loadUserByPubkey
  }
})

/**
 * Nostr Authentication Handler using nostr-login library
 * Implements NIP-46 (Nostr Connect) and other modern Nostr auth methods
 */

export class NostrAuth {
  constructor() {
    this.provider = 'nostr'
    this.isInitialized = false
    this.currentUser = null
    this.authListeners = new Set()
  }

  /**
   * Check if Nostr authentication is properly configured
   */
  isConfigured() {
    return typeof window !== 'undefined'
  }

  /**
   * Initialize nostr-login and set up event listeners
   */
  async initialize() {
    if (this.isInitialized) return

    try {
      // Wait for nostr-login to be available
      await this.waitForNostrLogin()
      
      // Set up auth event listener
      document.addEventListener('nlAuth', this.handleAuthEvent.bind(this))
      
      this.isInitialized = true
      console.log('✅ Nostr-login initialized successfully')
    } catch (error) {
      console.error('❌ Failed to initialize nostr-login:', error)
      throw error
    }
  }

  /**
   * Wait for nostr-login to be available
   */
  waitForNostrLogin(timeout = 10000) {
    return new Promise((resolve, reject) => {
      const startTime = Date.now()
      
      const checkNostrLogin = () => {
        if (window.nostr && document.dispatchEvent) {
          resolve()
        } else if (Date.now() - startTime > timeout) {
          reject(new Error('Nostr-login failed to load within timeout'))
        } else {
          setTimeout(checkNostrLogin, 100)
        }
      }
      
      checkNostrLogin()
    })
  }

  /**
   * Handle nostr-login auth events
   */
  handleAuthEvent(event) {
    console.log('🔔 Nostr auth event:', event.detail)
    
    const { type } = event.detail
    
    // Notify all listeners
    this.authListeners.forEach(listener => {
      try {
        listener(event.detail)
      } catch (error) {
        console.error('Auth listener error:', error)
      }
    })
  }

  /**
   * Add auth event listener
   */
  onAuthChange(callback) {
    this.authListeners.add(callback)
    
    // Return unsubscribe function
    return () => {
      this.authListeners.delete(callback)
    }
  }

  /**
   * Launch nostr-login authentication flow
   */
  async signIn(startScreen = 'welcome') {
    try {
      if (!this.isInitialized) {
        await this.initialize()
      }

      console.log('🚀 Launching nostr-login with screen:', startScreen)
      
      // Launch the nostr-login modal
      document.dispatchEvent(new CustomEvent('nlLaunch', { 
        detail: startScreen 
      }))

      // Return a promise that resolves when auth is complete
      return new Promise((resolve, reject) => {
        const timeout = setTimeout(() => {
          reject(new Error('Nostr authentication timeout'))
        }, 300000) // 5 minutes timeout

        const handleAuth = async (event) => {
          const { type } = event.detail
          
          if (type === 'login' || type === 'signup') {
            clearTimeout(timeout)
            document.removeEventListener('nlAuth', handleAuth)
            
            try {
              // Get complete user data including profile
              const userData = await this.getCurrentUser()
              resolve(userData)
            } catch (error) {
              reject(error)
            }
          }
        }

        document.addEventListener('nlAuth', handleAuth)
      })
    } catch (error) {
      console.error('❌ Nostr sign in error:', error)
      throw this.formatError(error)
    }
  }

  /**
   * Get current user profile from Nostr
   */
  async getCurrentUser() {
    try {
      if (!window.nostr) {
        throw new Error('Nostr not available')
      }

      const pubkey = await window.nostr.getPublicKey()
      
      // Get npub format
      const { nip19 } = await import('nostr-tools/nip19')
      const npub = nip19.npubEncode(pubkey)

      // Fetch profile metadata from Nostr network
      const profile = await this.fetchUserProfile(pubkey)

      return {
        pubkey,
        npub,
        method: 'nostr-login',
        profile: profile || {
          name: 'Nostr User',
          display_name: '',
          about: '',
          picture: null,
          website: null,
          nip05: null,
          lud16: null
        }
      }
    } catch (error) {
      console.error('❌ Failed to get current user:', error)
      throw error
    }
  }

  /**
   * Fetch user profile from Nostr network using NIP-01 (kind 0 events)
   */
  async fetchUserProfile(pubkey) {
    try {
      console.log('📡 Fetching Nostr profile for pubkey:', pubkey.slice(0, 8) + '...')
      
      const { SimplePool } = await import('nostr-tools/pool')
      const pool = new SimplePool()
      
      // Use multiple relays for better reliability
      const relays = [
        'wss://relay.damus.io',
        'wss://nos.lol',
        'wss://relay.snort.social',
        'wss://relay.nostr.band',
        'wss://nostr.wine'
      ]

      return new Promise((resolve) => {
        let profile = null
        let eventReceived = false
        
        const timeout = setTimeout(() => {
          if (sub) sub.close()
          pool.close(relays)
          console.log('⏰ Profile fetch timeout, using fallback')
          resolve(profile)
        }, 8000) // 8 second timeout

        const sub = pool.subscribeMany(relays, [
          { 
            kinds: [0], 
            authors: [pubkey], 
            limit: 1 
          }
        ], {
          onevent: (event) => {
            if (eventReceived) return // Only process first event
            eventReceived = true
            
            try {
              console.log('📄 Received profile event:', event)
              const profileData = JSON.parse(event.content)
              
              profile = {
                name: profileData.name || profileData.display_name || 'Nostr User',
                display_name: profileData.display_name || profileData.name || '',
                about: profileData.about || '',
                picture: profileData.picture || null,
                nip05: profileData.nip05 || null,
                website: profileData.website || null,
                lud16: profileData.lud16 || profileData.lud06 || null,
                banner: profileData.banner || null
              }
              
              console.log('✅ Profile parsed successfully:', profile)
              
              // Close subscription and resolve
              clearTimeout(timeout)
              if (sub) sub.close()
              pool.close(relays)
              resolve(profile)
              
            } catch (e) {
              console.warn('⚠️ Failed to parse profile data:', e)
              console.log('Raw event content:', event.content)
            }
          },
          oneose: () => {
            console.log('📡 End of stored events reached')
            if (!eventReceived) {
              // No profile found, resolve with null after a short delay
              setTimeout(() => {
                clearTimeout(timeout)
                if (sub) sub.close()
                pool.close(relays)
                resolve(profile)
              }, 2000)
            }
          },
          onclose: (reason) => {
            console.log('🔌 Subscription closed:', reason)
          }
        })
      })
    } catch (error) {
      console.error('❌ Failed to fetch user profile:', error)
      return null
    }
  }

  /**
   * Verify NIP-05 identifier
   */
  async verifyNip05(nip05) {
    if (!nip05 || !nip05.includes('@')) return false
    
    try {
      const { nip05: nip05Utils } = await import('nostr-tools/nip05')
      const profile = await nip05Utils.queryProfile(nip05)
      return !!profile
    } catch (error) {
      console.warn('NIP-05 verification failed:', error)
      return false
    }
  }

  /**
   * Sign out from nostr-login
   */
  async signOut() {
    try {
      console.log('🚪 Signing out from Nostr...')
      
      // Trigger nostr-login logout
      document.dispatchEvent(new Event('nlLogout'))
      
      this.currentUser = null
      
      console.log('✅ Nostr sign out successful')
      return { error: null }
    } catch (error) {
      console.error('❌ Nostr sign out error:', error)
      return { error: error.message }
    }
  }

  /**
   * Format error messages for better user experience
   */
  formatError(error) {
    const message = error.message || 'Nostr authentication failed'
    
    if (message.includes('timeout')) {
      return new Error('Authentication timeout. Please try again.')
    }
    if (message.includes('cancelled') || message.includes('denied')) {
      return new Error('Authentication was cancelled. Please try again.')
    }
    if (message.includes('not available')) {
      return new Error('Nostr authentication is not available. Please try again later.')
    }
    if (message.includes('failed to load')) {
      return new Error('Failed to load Nostr authentication. Please refresh the page and try again.')
    }
    
    return new Error(`Nostr authentication failed: ${message}`)
  }

  /**
   * Cleanup resources
   */
  cleanup() {
    if (this.isInitialized) {
      document.removeEventListener('nlAuth', this.handleAuthEvent.bind(this))
      this.authListeners.clear()
      this.isInitialized = false
    }
  }
}

export default NostrAuth

Nostr-Login

This library is a powerful window.nostr provider.

<script src='https://www.unpkg.com/nostr-login@latest/dist/unpkg.js'></script>

Just add the above script to your HTML and get a nice UI for users to login with Nostr Connect (nip46), with an extension, read-only login, account switching, OAuth-like sign up, etc. Your app just talks to the window.nostr, the rest is handled by nostr-login.

See it in action on nostr.band.
Options

You can set these attributes to the script tag to customize the behavior:

    data-dark-mode - true/false, default will use the browser's color theme
    data-bunkers - the comma-separated list of domain names of Nostr Connect (nip46) providers for sign up, i.e. nsec.app,highlighter.com
    data-perms - the comma-separated list of permissions requested by the app over Nostr Connect, i.e. sign_event:1,nip04_encrypt
    data-theme - color themes, one of default, ocean, lemonade, purple
    data-no-banner - if true, do not show the nostr-login banner, will need to launch the modals using event dispatch, see below
    data-methods - comma-separated list of allowed auth methods, method names: connect, extension, readOnly, local, all allowed by default.
    data-otp-request-url - URL for requesting OTP code
    data-otp-reply-url - URL for replying with OTP code
    data-title - title for the welcome screen
    data-description - description for the welcome screen
    data-start-screen - screen shown by default (banner click, window.nostr.* call), options: welcome, welcome-login, welcome-signup, signup, local-signup, login, otp, connect, login-bunker-url, login-read-only, connection-string, switch-account, import
    data-signup-relays - comma-separated list of relays where nip65 event will be published on local signup
    data-outbox-relays - comma-separated list of relays that will be added to nip65 event on local signup
    data-signup-nstart - "true" to use start.njump.me instead of local signup
    data-follow-npubs - comma-separated list of npubs to follow if njump.me signup is used

Example:

<script src='https://www.unpkg.com/nostr-login@latest/dist/unpkg.js' data-perms="sign_event:1,sign_event:0" data-theme="ocean"></script>

Updating the UI

Whenever user performs an auth-related action using nostr-login, a nlAuth event will be dispatched on the document, which you can listen to in order to update your UI (show user profile, etc):

document.addEventListener('nlAuth', (e) => {
  // type is login, signup or logout
  if (e.detail.type === 'login' || e.detail.type === 'signup') {
    onLogin();  // get pubkey with window.nostr and show user profile
  } else {
    onLogout()  // clear local user data, hide profile info 
  }
})

Launching, logout, etc

The nostr-login auth modals will be automatically launched whenever you make a call to window.nostr if user isn't authed yet. However, you can also launch the auth flow by dispatching a custom nlLaunch event:

document.dispatchEvent(new CustomEvent('nlLaunch', { detail: 'welcome' }));

The detail event payload can be empty, or can be one of welcome, signup, login, login-bunker-url, login-read-only, switch-account.

To trigger logout in the nostr-login, you can dispatch a nlLogout event:

document.dispatchEvent(new Event("nlLogout"));

To change dark mode in the nostr-login, you can dispatch a nlDarkMode event, with detail as darkMode boolean:

document.dispatchEvent(new CustomEvent("nlDarkMode", { detail: true }));

Use as a package

Install nostr-login package with npm and then:

import { init as initNostrLogin } from "nostr-login"

// make sure this is called before any
// window.nostr calls are made
initNostrLogin({/*options*/})

Now the window.nostr will be initialized and on your first call to it the auth flow will be launched if user isn't authed yet.

You can also launch the auth flow yourself:

import { launch as launchNostrLoginDialog } from "nostr-login"

// make sure init() was called 

// on your signup button click
function onSignupClick() {
  // launch signup screen
  launchNostrLoginDialog({
    startScreen: 'signup'
  })
}

Next.js Fix for Server Side Rendering (SSR)

nostr-login calls document which is unavailable for server-side rendering. You will have build errors. To fix this, you can import nostr-login on the client side in your component with a useEffect like this:

  useEffect(() => {
    import('nostr-login')
      .then(async ({ init }) => {
        init({
          // options
        })
      })
      .catch((error) => console.log('Failed to load nostr-login', error));
  }, []);

Note: even if your component has "use client" in the first line, this fix still may be necessary.

API:

    init(opts) - set mapping of window.nostr to nostr-login
    launch(startScreen) - launch nostr-login UI
    logout() - drop the current nip46 connection

Options:

    theme - same as data-theme above
    startScreen - same as startScreen for nlLaunch event above
    bunkers - same as data-bunkers above
    devOverrideBunkerOrigin - for testing, overrides the bunker origin for local setup
    onAuth: (npub: string, options: NostrLoginAuthOptions) - a callback to provide instead of listening to nlAuth event
    perms - same as data-perms above
    darkMode - same as data-dark-mode above
    noBanner - same as data-no-banner above
    isSignInWithExtension - true to bring the Sign in with exception button into main list of options, false to hide to the Advanced, default will behave as true if extension is detected.
